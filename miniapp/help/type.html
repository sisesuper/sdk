# 类型

# Config

初始化连接配置项

type Config = {
  interceptor?: Interceptor;
  request?: AxiosRequestConfig;
  component?: AppComponentConfig;
};

  • interceptor?: 组件请求地址

    • 参考：Interceptoropen in new window

  • request?： Axios封装的请求配置

    • 参考：Axiosopen in new window

  • component?：组件配置

    • 参考：AppComponentConfigopen in new window

# AppComponentConfig

组件请求地址

type AppComponentConfig = {
  host?: string;
  prefix?: string;
};

  • host?：组件请求地址

  • prefix?：组件请求地址前缀

# Interceptor

代理请求

type Interceptor = {
  request?: {
    handle: InterceptorRequest;
    error?: InterceptorError;
  };
  response?: {
    handle: InterceptorResponse;
    error?: InterceptorError;
  };
};

# AppOptions

type AppOptions = {
  flex?: Flex;
  version: string;
  name: string;
  mode: Mode;
  theme?: ThemeOptions; // 主题风格
  defaultPath?: string; // 默认的路由path
  component?: ComponentOptions;
};

  • flex?: 应用弹性的方式

    • 参考：Flexopen in new window

  • version: 版本号

  • name: 名称

  • mode: 模式

    • 参考：Modeopen in new window

  • theme?: 主题配置

    • 参考：ThemeOptionsopen in new window

  • defaultPath?: 默认的路由 path

  • component?: 组件

    • 参考：ComponentOptionsopen in new window

# Flex

enum Flex {
  auto = "auto", // 自动
  zoom = "zoom", // 自动缩放
  full = "full", // 铺满全屏
  horizontal = "horizontal", // 水平方向自动，垂直方向撑满
  vertical = "vertical", // 垂直方向自动，水平方向撑满
}

# Mode

type Mode = "edit" | "preview"; // 编辑 | 预览

# ThemeOptions

interface ThemeOptions {
  name: string; // 主题key
  zhname: string; // 主题名称
  style: {
    [key: string]: string;
  };
}

# ComponentOptions

type ComponentOptions = {
  version: string; // 组件版本号
  label?: string; // 组件中文名，用于编辑器显示
  name: string; // 组件
  key: string; // 组件key
  uid: string; // 组件的唯一ID
  visible: boolean; // 组件是否可见
  detach: boolean; // 是否脱离文档流
  locked?: boolean; // 锁定状态
  style?: Partial<CSSStyleDeclaration>; // 动态配置的样式
  props?: Record<string, any>; // 组件的默认参数
  attribute?: Record<string, any>; // 配置的属性
  source?: Array<Source>; // 数据源
  animation?: Array<AnimationOptions>; // 组件动画
  interaction?: InterAction[]; // 交互
  scripts?: Script[]; // 动态脚本
  compiled?: string; // 编译后可在浏览器端直接运行的脚本
  children?: Options[];
  layers?: LayerOptions[];
};

  • source: 数据源

    • 参考：Sourceopen in new window

  • animation: 组件动画

    • 参考：AnimationOptionsopen in new window

  • interaction: 交互

    • 参考：InterActionopen in new window

  • scripts: 动态脚本

    • 参考：Scriptopen in new window

  • layers: 图层

    • 参考：LayerOptionsopen in new window

# Source

type Source = {
  key: string;
  name?: string; // 数据的名称
  from: string;
  data: any;
};

# AnimationOptions

type AnimationOptions = {
  driver?: AnimationDriver;
  key: string; // 动画标识
  value: string | string[]; // 动画值，如：animate.css 的 bounceInLeft
  options?: {
    status?: boolean; // 是否启用
    delay?: string; // 延迟时间
    duration?: string; // 动画时长
    repeat?: string; // 重复次数
    toward?: string; // 运动方向
  } & Record<string, any>;
};

  • driver

    • 参考：AnimationDriveropen in new window

# AnimationDriver

type AnimationDriver = "css" | "tween";

# InterAction

type InterAction = {
  uid: string; // 交互唯一ID
  name: string; // 交互名称
  eventKey: string; // 事件key
  type: "script" | "action";
  actions: Array<{
    key: ActionKey | ScriptKey;
    targets?: string[];
    params?: Record<string, any>;
  }>; // 动作类型对应操作的对象ID数组
};

  • action:动作

    • 参考：ActionKeyopen in new window

    • 参考：ScriptKeyopen in new window

# LayerOptions

type LayerOptions = {
  uid: string; // 图层唯一ID（自动生成）
  key: string; // 图层的标识
  name: string; // 图层名称
  defaultRender: boolean; // 是否默认渲染
  props?: Record<string, any>;
  component: Options;
};

  • component: 组件

    • 参考：ComponentOptionsopen in new window

# LayerItem

type LayerItem = LayerOptions & {
  instance: Component | undefined
};

# ComponentType

type ComponentType = "component" | "page" | "layer" | "group";

# AppLifeCycle

enum AppLifeCycle {
  beforeLoading = "beforeLoading",
  loaded = "loaded",
  beforeRender = "beforeRender",
  mounted = "mounted",
  beforeDestroy = "beforeDestroy",
  destroyed = "destroyed",
}

# ComponentLifeCycle

enum ComponentLifeCycle {
  beforeCreate = "beforeCreate",
  created = "created",
  mounted = "mounted",
  render = "render",
  beforeDestroy = "beforeDestroy",
  destroyed = "destroyed",
}

# Action

type Action = {
  key: ActionKey; // key
  desc: string; // 描述
  method: <P extends Record<string, any>>(params?: P) => void;
};

# ActionKey

type ActionKey = string;

# Script

type Script = {
  uid: string; // 脚本id
  key: ScriptKey; // 脚本名称
  content: string; // 代码片段
  type: ScriptType; // 代码的作用：联动、插件、场景
};

# ScriptKey

type ActionKey = string;

# ScriptType

type ScriptType = "action" | "plugin" | "scene";

# AlignType

enum AlignType {
  VerticalStart = "VerticalStart",
  VerticalMiddle = "VerticalMiddle",
  VerticalEnd = "VerticalEnd",
  HorizontalStart = "HorizontalStart",
  HorizontalMiddle = "HorizontalMiddle",
  HorizontalEnd = "HorizontalEnd",
}

# AutoOptions

type AutoOptions = {
  width?: number; // 布局的宽度
  height?: number; // 布局的高度
  gutters?: number[]; // 间隔
};

# ComponentEvent

type ComponentEvent = {
  from: string; // 组件ID
  key: string; // 事件标识
  type?: string; // 事件类型
  name: string; // 事件名称
  desc?: string; // 事件描述
  global?: boolean; // 是把当前组件的事件暴露到全局
};
